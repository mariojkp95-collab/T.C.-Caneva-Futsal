// Script to fetch data from Airtable and save it locally const AIRTABLE_CONFIG = { token: 'YOUR_TOKEN_HERE', baseId: 'appakHgEWKsk7IWNK', tables: { giocatori: 'tblr9VNukwxd6LAuU', partite: 'tbl5Kii1jfpoQHxSr', allenamenti: 'tbleGoRCEQNGWyh32', gol: 'tblkzYPzyZXgUpb6Q' } }; async function fetchAirtableData() { const baseUrl = `https://api.airtable.com/v0/${AIRTABLE_CONFIG.baseId}`; const headers = { 'Authorization': `Bearer ${AIRTABLE_CONFIG.token}`, 'Content-Type': 'application/json' }; try { console.log('Fetching data from Airtable...'); const [giocatoriRes, partiteRes, allenamentiRes, golRes] = await Promise.all([ fetch(`${baseUrl}/${AIRTABLE_CONFIG.tables.giocatori}`, { headers }), fetch(`${baseUrl}/${AIRTABLE_CONFIG.tables.partite}`, { headers }), fetch(`${baseUrl}/${AIRTABLE_CONFIG.tables.allenamenti}`, { headers }), fetch(`${baseUrl}/${AIRTABLE_CONFIG.tables.gol}`, { headers }) ]); const giocatori = await giocatoriRes.json(); const partite = await partiteRes.json(); const allenamenti = await allenamentiRes.json(); const gol = await golRes.json(); console.log('Raw data from Airtable:'); console.log('Giocatori:', giocatori); console.log('Partite:', partite); console.log('Allenamenti:', allenamenti); console.log('Gol:', gol); // Parse data const parsedData = { players: parseGiocatori(giocatori.records, gol.records), events: [ ...parsePartite(partite.records), ...parseAllenamenti(allenamenti.records) ] }; console.log('\n\n=== DATI PRONTI PER IL SITO ===\n'); console.log('Copia questo JSON e sostituiscilo nel file js/data.js:\n'); console.log(JSON.stringify(parsedData, null, 2)); return parsedData; } catch (error) { console.error('Error fetching data:', error); } } function parseGiocatori(giocatori, gol) { // Count goals per player const goalCounts = {}; const assistCounts = {}; gol.forEach(record => { const fields = record.fields; // Count goals if (fields.Giocatore && fields.Giocatore.length > 0) { const playerId = fields.Giocatore[0]; goalCounts[playerId] = (goalCounts[playerId] || 0) + 1; } // Count assists if (fields.Assist && fields.Assist.length > 0) { const assistId = fields.Assist[0]; assistCounts[assistId] = (assistCounts[assistId] || 0) + 1; } }); return giocatori.map(record => { const fields = record.fields; return { id: record.id, firstName: fields.Nome || '', lastName: fields.Cognome || '', number: fields.Numero || 0, position: fields.Ruolo || '', goals: goalCounts[record.id] || 0, assists: assistCounts[record.id] || 0, motm: fields.MOTM || 0 }; }); } function parsePartite(partite) { return partite.map(record => { const fields = record.fields; return { id: record.id, type: 'match', title: fields.Avversario ? `TC Caneva vs ${fields.Avversario}` : 'Partita', date: fields.Data || '', time: fields.Ora || '', location: fields.Luogo || '', opponent: fields.Avversario || '', result: fields.Risultato || null }; }); } function parseAllenamenti(allenamenti) { return allenamenti.map(record => { const fields = record.fields; return { id: record.id, type: 'training', title: fields.Titolo || 'Allenamento', date: fields.Data || '', time: fields.Ora || '', location: fields.Luogo || '' }; }); } // Run the fetch fetchAirtableData();
